<script>
  // Data mahasiswa (contoh)
  const mahasiswaData = [
    { nik: '1234567890123456', nama: 'Andi Wijaya', nomor: 'MB2023001', prodi: 'Teknik Informatika', status: 'Lulus' },
    { nik: '2345678901234567', nama: 'Budi Santoso', nomor: 'MB2023002', prodi: 'Manajemen', status: 'Tidak Lulus' },
    { nik: '3456789012345678', nama: 'Citra Dewi', nomor: 'MB2023003', prodi: 'Akuntansi', status: 'Lulus' },
    { nik: '4567890123456789', nama: 'Dewi Lestari', nomor: 'MB2023004', prodi: 'Sistem Informasi', status: 'Lulus' },
    { nik: '5678901234567890', nama: 'Eko Prasetyo', nomor: 'MB2023005', prodi: 'Ekonomi', status: 'Tidak Lulus' }
  ];

  const loginCard = document.getElementById('loginCard');
  const resultCard = document.getElementById('resultCard');
  const errorMsg = document.getElementById('errorMsg');
  const resultContent = document.getElementById('resultContent');
  const nikInput = document.getElementById('nikInput');

  function login() {
    const nik = nikInput.value.trim();

    errorMsg.textContent = '';

    if (nik.length !== 16 || !/^\d{16}$/.test(nik)) {
      errorMsg.textContent = 'NIK harus 16 digit angka.';
      return;
    }

    const mahasiswa = mahasiswaData.find(m => m.nik === nik);

    if (!mahasiswa) {
      errorMsg.textContent = 'NIK tidak ditemukan. Periksa kembali dan coba lagi.';
      return;
    }

    // Tampilkan data mahasiswa
    loginCard.style.display = 'none';
    resultCard.style.display = 'block';

    const statusClass = mahasiswa.status.toLowerCase().replace(' ', '-');

    resultContent.innerHTML = `
      <p><strong>Nama:</strong> ${mahasiswa.nama}</p>
      <p><strong>Nomor Pendaftaran:</strong> ${mahasiswa.nomor}</p>
      <p><strong>Program Studi Pilihan:</strong> ${mahasiswa.prodi}</p>
      <p><strong>Status Kelulusan:</strong> <span class="${statusClass}">${mahasiswa.status}</span></p>
    `;
  }

  function logout() {
    nikInput.value = '';
    errorMsg.textContent = '';
    resultContent.innerHTML = '';
    resultCard.style.display = 'none';
    loginCard.style.display = 'block';
  }
</script>
